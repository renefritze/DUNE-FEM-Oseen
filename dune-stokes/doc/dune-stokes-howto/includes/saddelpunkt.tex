
\section{Sattelpunktlösungsalgorithmus}
\begin{itemize}
 \begin{item}{Wähle}
	\begin{align*}
		p^0 = 0	 
	\end{align*}
 \end{item}
 \begin{item}{Löse}
	\begin{align*}
	 	u^0 &= A^{-1} ( F - B p^0)
	\end{align*}
 \end{item} 
 \begin{item}{Setze}
	\begin{align*}
	 	r^0 &= G - B_T u^0 + C p^0 \\
		d^0 &= r^0 \\
		\delta^0 &= < r^0, r^0 >
	\end{align*}
 \end{item} 
 \begin{item}{Wiederhole:}
	\begin{itemize}
		\begin{item}{Löse:}
			\begin{align*}
				\chi^m &= A^{-1} B d^m \\
			\end{align*}
		\end{item} 
		\begin{item}{setze:}
			\begin{align*}
				h^m &= B_T \chi^m + C d^m \\
				\rho^m &= \delta^m / < h^m,d^m> \\
				p^{m+1} &= p^m - \rho^m d^m \\
				u^{m+1} &= u^m + \rho^m \chi^m \\
				r^{m+1} &= r^m - \rho^m h^m \\
				\delta^{m+1} &= < r^{m+1} , r^{m+1} >
			\end{align*}
		\end{item} 
		\begin{item}
			{falls $\delta^{m+1} < \epsilon $ STOP}
		\end{item} 
		\begin{item}{setze:}
			\begin{align*}
				\gamma^m &= \delta^{m+1} / \delta^m \\
				d^{m+1} &= r^{m+1} + \gamma^m d^m
			\end{align*}
		\end{item}
	\end{itemize}
\end{item}
\end{itemize}
